Classes

// Les classes permettent de créer des templates pour un concept abstrait qui va être utilisé plusieurs fois.

class Person {                               // on initialise la classe 'Person'
  constructor(name, age) {                   // 'constructor' est une méthode spéciale qui s'éxécute lors de la création d'une instance de la classe
    this.name = name;                        // 'this' est un mot-clé qui substitue l'instance lors de l'initialisation de la classe (ça a l'air dur mais tkt)
    this.age = age;                          // la classe 'Person' a donc 2 attributs, 'name' et 'age'
  }

  greet = () => {                            // on peut créer des méthodes dans des classes, qui pourront être utilisées par les instances de cette classe
    console.log(`Hello, ${this.name}!`);     // ici, on pourra voir 'Hello, <instance.name>!' <instance.name> étant le nom de l'instance en question
  }
}

let martin = new Person("Martin", 20);       // on crée une nouvelle instance de la classe 'Person', et on remplit les paramètres comme si l'on remplissait la fonction 'constructor'

console.log(martin.name);    // Martin           // l'instance a donc pour name "Martin"
console.log(martin.age);     // 20               // et pour âge 20
martin.greet();              // Hello, Martin!   // on peut éxécuter sa méthode

// le mot-clé 'this' prend donc la forme de 'martin'
// En réutilisant le code ci-dessus, on peut créer une nouvelle instance de classe.

let literally_me = new Person("Guts", 24);

console.log(literally_me.name);    // Guts
console.log(literally_me.age);     // 24
literally_me.greet()               // Hello, Guts!

// On comprend donc que dans le premier cas, le mot-clé 'this' prend la forme de 'martin', mais dans le cas actuel, il prend la forme de 'literally_me'
// 'this' représente en fait l'objet en question
// ATTENTION: le mot-clé 'this' ne peut être utilisé que dans déclaration de classe, ça serait pas logique sinon
